00054 bool Keyboard_Controller::key_decoded () {
00055   bool done = false;
00056 
00057   // keys of MF II keyboards start sending one of two prefix bytes
00058   if (code == prefix1 || code == prefix2) { 
00059     prefix = code;
00060     return false;
00061   }
00062 
00063   // releasing a key is important for the modifier keys (SHIFT, CTRL, ALT, ...)
00064   // break codes from other keys are ignored
00065   if (code & break_bit) {
00066     // break code of a key is equal to the make code but additionaly the 
00067     // break_bit is set
00068     code &= ~break_bit;
00069     switch (code) {
00070       case 42:
00071       case 54:
00072         gather.shift (false);
00073         break;
00074       case 56:
00075         if (prefix == prefix1)
00076           gather.alt_right (false);
00077         else
00078           gather.alt_left (false);
00079         break;
00080       case 29:
00081         if (prefix == prefix1)
00082           gather.ctrl_right (false);
00083         else
00084           gather.ctrl_left (false);
00085         break;
00086     }
00087 
00088     // a prefix is only valid for the following code
00089     prefix = 0;
00090 
00091     // returns false since the break codes gives no additional information
00092     return false;
00093   }
00094 
00095   // a key has been pressed, if it was a modifier key like SHIFT, ALT, NUM_LOCK
00096   // eg. only the internal state is changed. Return value 'false' shows that 
00097   // the input is not finished yet. If other keys have been pressed the ASCII 
00098   // and Scancode is read and 'ture' is returned.
00099   switch (code) {
00100     case 42:
00101     case 54:
00102       gather.shift (true);
00103       break;
00104     case 56:
00105       if (prefix == prefix1)
00106         gather.alt_right (true);
00107       else
00108         gather.alt_left (true);
00109         break;
00110     case 29:
00111       if (prefix == prefix1)
00112         gather.ctrl_right (true);
00113       else
00114         gather.ctrl_left (true);
00115       break;
00116     case 58:
00117       gather.caps_lock (!gather.caps_lock ());
00118       set_led (caps_lock, gather.caps_lock());
00119       break;
00120     case 70:
00121       gather.scroll_lock (!gather.scroll_lock ());
00122       set_led (scroll_lock, gather.scroll_lock());
00123       break;
00124     case 69: // NUM_LOCK or BREAK ?
00125       if (gather.ctrl_left ()) {        // BREAK
00126         get_ascii_code ();
00127         done = true;
00128       } else {                          // NUM_LOCK
00129         gather.num_lock (!gather.num_lock());
00130         set_led (num_lock, gather.num_lock ());
00131       }
00132       break;
00133     default:                            // other keys
00134        // get ASCII codes from table, DONE
00135       get_ascii_code ();
00136       done = true;
00137   }
00138 
00139     // a prefix is only valid for the following code
00140   prefix = 0;
00141 
00142   if (done)
00143     return true;
00144   else
00145     return false;
00146 }

00151 void Keyboard_Controller::get_ascii_code () {
00152   if (code == 53 && prefix == prefix1) { // special case scancode 53
00153     gather.ascii ('/');
00154     gather.scancode (Key::scan::div);
00155   } else if (gather.num_lock () && !prefix && code>=71 && code<=83) {
00156     gather.ascii (asc_num_tab[code-71]);
00157     gather.scancode (scan_num_tab[code-71]);
00158   } else if (gather.alt_right ()) {
00159     gather.ascii (alt_tab[code]);
00160     gather.scancode (code);
00161   } else if (gather.shift ()) {
00162     gather.ascii (shift_tab[code]);
00163     gather.scancode (code);
00164   } else if (gather.caps_lock ()) {
00165     if ((code>=16 && code<=26) || (code>=30 && code<=40)
00166         || (code>=44 && code<=50)) {
00167       gather.ascii (shift_tab[code]);
00168       gather.scancode (code);
00169     } else {
00170       gather.ascii (normal_tab[code]);
00171       gather.scancode (code);
00172     }
00173   } else {
00174     gather.ascii (normal_tab[code]);
00175     gather.scancode (code);
00176   }
00177 }
